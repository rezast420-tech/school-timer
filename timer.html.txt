<script>
let timer;
function startTimer() {
    clearInterval(timer);
    let sec = parseInt(document.getElementById("seconds").value);
    if (isNaN(sec) || sec <= 0) { alert("????? ??? ????? ???? ????."); return; }
    updateDisplay(sec);
    timer = setInterval(() => {
        sec--;
        if (sec < 0) { clearInterval(timer); alert("???? ???? ??!"); return; }
        updateDisplay(sec);
    }, 1000);
}

function updateDisplay(sec) {
    let m = Math.floor(sec / 60);
    let s = sec % 60;
    document.getElementById("countdown").textContent = `${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
}
